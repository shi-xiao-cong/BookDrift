# 使用 Java 17 基础镜像
FROM openjdk:17-jdk-slim

# 设置工作目录
WORKDIR /app

# 复制 Maven 构建文件
COPY mvnw .
COPY .mvn .mvn
COPY pom.xml .

# 下载依赖
RUN chmod +x mvnw
RUN ./mvnw dependency:go-offline -B

# 复制源代码
COPY src src

# 打包应用
RUN ./mvnw clean package -DskipTests

# 复制生成的 jar 包
RUN cp target/*.jar app.jar

# 暴露端口
EXPOSE 10011

# 启动应用
ENTRYPOINT ["java", "-jar", "-Dspring.profiles.active=railway", "app.jar"]
